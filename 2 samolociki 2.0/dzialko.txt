void ustawSie(object obiekt){
	point miejsce = space(new point(obiekt.position.x,obiekt.position.y,obiekt.position.z),25,50,2);
	goto(miejsce,70);
	aim(0,direction(obiekt.position));
}

void zabij(object kogo){
	ustawSie(kogo);
	fire(0.1);
}


void sprawdzEnergie(object kogo,object gdzie,float poziom){
	if(kogo.energyCell.energyLevel<poziom){
		goto(gdzie.position,70);
		wait(5);
	}
}

{
	  object stacja = radar(PowerStation);
